\begin{frame}{Access modifiers (quick table)}
  \begin{center}
  \begin{tabular}{ll}
    \toprule
    \textbf{Modifier} & \textbf{Visible where?} \\
    \midrule
    \texttt{private} & inside the same class \\
    (default) & inside the same package \\
    \texttt{protected} & same package + subclasses \\
    \texttt{public} & everywhere \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3em}
  \textbf{Rule of thumb:} fields are usually \texttt{private}.
  \slidesources{\OOPSUnitIISources}
\end{frame}

\begin{frame}{Packages: why and how}
  \begin{itemize}[<+->]
    \item Organize code and avoid name conflicts.
    \item First line in file: \texttt{package com.upes.demo.pay;}
    \item Folder matches package: \texttt{com/upes/demo/pay/}
    \item Use \texttt{import} to avoid long names.
  \end{itemize}
  \slidesources{\OOPSUnitIISources}
\end{frame}

\begin{frame}{Interfaces (intro)}
  \begin{itemize}[<+->]
    \item Interface = \textbf{contract} (what operations must exist).
    \item Classes \textbf{implement} interfaces.
    \item Benefit: reduce coupling; caller depends on interface, not concrete class.
  \end{itemize}
  \slidesources{\OOPSUnitIISources}
\end{frame}


\section{Demo / Example}
\label{sec:demo}

\begin{frame}[fragile]{Example: interface contract}
\begin{lstlisting}
public interface Payable {
    double amountDue();
}

public class Invoice implements Payable {
    private final double total;
    public Invoice(double total) { this.total = total; }
    @Override public double amountDue() { return total; }
}
\end{lstlisting}
  \slidesources{\OOPSUnitIISources}
\end{frame}
