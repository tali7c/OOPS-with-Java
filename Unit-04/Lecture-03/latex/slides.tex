\begin{frame}{Key Terms to Know}
\begin{itemize}[<+->]
  \item \textbf{Functional interface}: interface with 1 abstract method (SAM).
  \item \textbf{SAM}: Single Abstract Method.
  \item \textbf{Lambda}: expression like \texttt{(a, b) -> a + b}.
  \item \textbf{Target typing}: compiler infers lambda type from context (assignment/argument).
  \item \textbf{Effectively final}: local variable not modified after initialization (can be captured in lambda).
\end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Functional interface (SAM)}
  \begin{itemize}[<+->]
    \item Interface with exactly one abstract method.
    \item Examples: \texttt{Runnable}, \texttt{Comparator<T>}, many Swing listeners.
    \item Optional: annotate with \texttt{@FunctionalInterface}.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Target typing (why lambdas work)}
  \begin{itemize}[<+->]
    \item Lambda gets its type from context (assignment or method parameter).
    \item Example: \texttt{Runnable r = () -> ...}
    \item Example: \texttt{list.sort((a,b) -> ...)}
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Anonymous Class $\rightarrow$ Lambda}
  \begin{itemize}[<+->]
    \item Start from a one-method anonymous class that already matches a functional interface.
    \item Keep the parameter list and method body; drop the class boilerplate.
    \item Result: shorter code, same contract, easier to read when the behavior is small.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Lambda syntax (common forms)}
  \begin{itemize}[<+->]
    \item \texttt{() -> expr}
    \item \texttt{x -> expr}
    \item \texttt{(a, b) -> expr}
    \item \texttt{x -> \{ ...; return ...; \}}
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Readability guidelines}
  \begin{itemize}[<+->]
    \item Use lambdas for short, clear behavior (1--3 lines).
    \item Avoid long lambdas; extract to a named method.
    \item Use meaningful parameter names if not obvious.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\section{Summary}

\begin{frame}{Key Takeaways}
