\begin{frame}{Motivation: the problem without generics}
  \begin{itemize}[<+->]
    \item If you store values as \texttt{Object}, you lose type information.
    \item You must cast back to the expected type.
    \item Wrong cast $\rightarrow$ \texttt{ClassCastException} at runtime.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{What generics give you}
  \begin{itemize}[<+->]
    \item Compile-time type checking (errors caught early).
    \item Less casting and cleaner code.
    \item Better readability: \texttt{List<String>} clearly states element type.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Generic class idea}
  \begin{itemize}[<+->]
    \item Write a class once: \texttt{Box<T>}
    \item Use it many times: \texttt{Box<String>}, \texttt{Box<Integer>}, \texttt{Box<Student>}
    \item The compiler enforces types for each use.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Raw types (avoid)}
  \begin{itemize}[<+->]
    \item Raw type: \texttt{Box} instead of \texttt{Box<T>}.
    \item Compiles with warnings but removes type safety.
    \item Prefer \texttt{Box<?>} when type is unknown.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}


\section{Demo / Example}
\label{sec:demo}

\begin{frame}[fragile]{Example: generic class}
\begin{lstlisting}
class Box<T> {
    private final T value;
    Box(T value) { this.value = value; }
    T get() { return value; }
}
\end{lstlisting}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}[fragile]{Example usage}
\begin{lstlisting}
public class Demo {
    public static void main(String[] args) {
        Box<String> b1 = new Box<>("UPES");
        String s = b1.get(); // no cast

        Box<Integer> b2 = new Box<>(10);
        int x = b2.get();
        System.out.println(s + " " + x);
    }
}
\end{lstlisting}
  \slidesources{\OOPSUnitIVSources}
\end{frame}
