\begin{frame}{Key Terms to Know}
\small
\begin{columns}[t]
\column{0.48\textwidth}
\begin{itemize}
  \item \textbf{Event source}: UI component that triggers an event (button, menu item).
  \item \textbf{Event object}: information about the event (e.g., \texttt{ActionEvent}).
  \item \textbf{Listener}: object/function that handles events (e.g., \texttt{ActionListener}).
  \item \textbf{EDT}: Event Dispatch Thread (Swing UI thread).
  \item \textbf{JDBC}: Java Database Connectivity API.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}
  \item \textbf{Connection}: active DB connection.
  \item \textbf{PreparedStatement}: parameterized SQL statement (safer and common).
  \item \textbf{ResultSet}: table-like cursor over query results.
  \item \textbf{CRUD}: Create, Read, Update, Delete.
  \item \textbf{DAO}: Data Access Object (separate DB code from UI).
\end{itemize}
\end{columns}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Swing events (model)}
  \begin{itemize}[<+->]
    \item Source: component (e.g., \texttt{JButton})
    \item Event: object with details (e.g., \texttt{ActionEvent})
    \item Listener: handler (e.g., \texttt{ActionListener})
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Registering a listener}
  \begin{itemize}[<+->]
    \item \texttt{saveBtn.addActionListener(...);}
    \item In Java 8+, a lambda is commonly used for \texttt{ActionListener}.
    \item Keep handler short: validate input, call DAO, show message.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{JDBC flow (memorize)}
  \begin{enumerate}[<+->]
    \item Get \texttt{Connection}
    \item Create \texttt{PreparedStatement}
    \item Execute SQL (\texttt{executeUpdate}/\texttt{executeQuery})
    \item Read \texttt{ResultSet} for SELECT
    \item Close resources (use \texttt{try-with-resources})
  \end{enumerate}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{Statement vs PreparedStatement}
  \begin{itemize}[<+->]
    \item \texttt{Statement}: builds SQL strings; risky with user input.
    \item \texttt{PreparedStatement}: parameterized (\texttt{?}); safer and common.
    \item \texttt{CallableStatement}: stored procedures (DB-specific).
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}{ResultSet (reading rows)}
  \begin{itemize}[<+->]
    \item Use cursor loop: \texttt{while (rs.next()) \{...\}}
    \item Get columns: \texttt{rs.getString("name")}, \texttt{rs.getInt("capacity")}
    \item Map to objects: \texttt{Resource} model class
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\section{Demo / Example}
\label{sec:demo}

\begin{frame}[fragile,shrink=8]{Example: button click handler (save)}
\begin{lstlisting}
saveBtn.addActionListener(e -> {
    String name = nameField.getText().trim();
    if (name.isEmpty()) {
        JOptionPane.showMessageDialog(f, "Name required");
        return;
    }
    saveInBackground(name, type, capacity);
});
\end{lstlisting}
  \begin{itemize}[<+->]
    \item Listener does quick validation only.
    \item A helper/background task performs the DB insert and shows success/failure.
  \end{itemize}
  \slidesources{\OOPSUnitIVSources}
\end{frame}

\begin{frame}[fragile]{Example: JDBC insert (PreparedStatement)}
\begin{lstlisting}
String sql = "INSERT INTO resources(name, type, capacity) VALUES(?,?,?)";
try (Connection con = DriverManager.getConnection(url, user, pass);
     PreparedStatement ps = con.prepareStatement(sql)) {
  ps.setString(1, r.getName());
  ps.setString(2, r.getType());
  ps.setInt(3, r.getCapacity());
  ps.executeUpdate();
}
\end{lstlisting}
  \slidesources{\OOPSUnitIVSources}
\end{frame}
