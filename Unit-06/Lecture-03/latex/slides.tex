\begin{frame}{UML notation (quick cheat sheet)}
  \begin{itemize}[<+->]
    \item Class box: \textbf{Name} / \textbf{Attributes} / \textbf{Methods}
    \item Visibility: \texttt{+} public,\; \texttt{-} private,\; \texttt{\#} protected
    \item Relationships: association (line), inheritance (triangle), composition (filled diamond)
    \item Multiplicity: \texttt{1}, \texttt{0..1}, \texttt{*}, \texttt{1..*}
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{From use cases to classes}
  \begin{enumerate}[<+->]
    \item Use cases $\rightarrow$ nouns (entities) + verbs (operations)
    \item Entities: Resource, Booking, User
    \item Services: BookingService (rules)
    \item DAOs: BookingDao (DB)
    \item UI: panels/screens
  \end{enumerate}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Responsibilities (avoid God classes)}
  \begin{itemize}[<+->]
    \item UI: read inputs, show messages
    \item Service: validate rules (overlap checks)
    \item DAO: SQL + CRUD only
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Relationships (use carefully)}
  \begin{itemize}[<+->]
    \item Prefer composition (has-a) for most domain links.
    \item Use inheritance only for true is-a.
    \item Interfaces help reduce coupling (e.g., repository/DAO interface).
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Package structure (recommended)}
  \begin{itemize}[<+->]
    \item \texttt{com.upes.crm.model}
    \item \texttt{com.upes.crm.dao}
    \item \texttt{com.upes.crm.service}
    \item \texttt{com.upes.crm.ui}
    \item \texttt{com.upes.crm.util}
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}
