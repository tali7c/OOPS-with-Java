\begin{frame}{Today: Roadmap}

\begin{frame}{Capstone goal (what you will build)}
  \begin{itemize}[<+->]
    \item A small complete product: Swing UI + JDBC + OOP design.
    \item Evidence of teamwork: GitHub issues, branches, PRs.
    \item Clear scope and deliverables across Unit 06 lectures.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Scope boundaries (must do)}
  \begin{itemize}[<+->]
    \item Must-have: required for minimum demo.
    \item Should-have: good improvements if time permits.
    \item Could-have: optional enhancements.
    \item Non-goals: explicitly out of scope.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Team setup checklist}
  \begin{enumerate}[<+->]
    \item Create repo + add collaborators
    \item Add README (project + run steps + team)
    \item Create labels (feature, bug, ui, db, doc)
    \item Create initial issues (8--12)
  \end{enumerate}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}[fragile]{Suggested repo skeleton (keep it simple)}
\begin{lstlisting}[language=]
repo-root/
  README.md
  docs/   (deliverables: D0, D1, D2...)
  src/    (Java packages)
  db/     (schema.sql)
\end{lstlisting}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Git workflow (simple)}
  \begin{itemize}[<+->]
    \item Keep \texttt{main} stable.
    \item Work on feature branches: \texttt{feature/<name>}.
    \item Open PRs, review, then merge.
    \item Use clear commit messages.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{What a good GitHub issue includes}
  \begin{itemize}[<+->]
    \item Clear goal (one feature or one bug fix)
    \item Acceptance criteria (3--6 testable bullets)
    \item Task checklist (UI/service/DAO/DB/test/doc)
    \item Label + assignee + link to PR (when created)
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Deliverable-0 checklist}
  \begin{itemize}[<+->]
    \item 1-page proposal: title, problem, Must/Should/Could, screens, DB tables, milestones, risks, repo link.
    \item Backlog: issues for must-have features + technical tasks.
    \item Evidence: at least 1 PR created (even small setup PR).
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}
