\begin{frame}{Requirements: weak vs testable}
  \textbf{Weak (avoid):}
  \begin{itemize}
    \item ``The system should be fast.''
    \item ``UI should be user friendly.''
  \end{itemize}
  \vspace{0.4em}
  \textbf{Better (testable):}
  \begin{itemize}
    \item ``Booking cannot overlap an existing booking for the same resource.''
    \item ``On success, record is saved in DB and appears in list view.''
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Find actors and goals}
  \begin{itemize}[<+->]
    \item Actors = roles outside the system (Admin, User, Staff).
    \item Each actor has goals $\rightarrow$ use cases.
    \item Start from Must-have scope.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Use case writing template}
  \begin{itemize}[<+->]
    \item Name + primary actor
    \item Preconditions + trigger
    \item Main success steps (5--10 steps)
    \item Alternate flows (failures/edge cases)
    \item Postconditions
    \item Acceptance criteria (testable)
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Mini use case: Create Booking (excerpt)}
  \textbf{Main success scenario (sample):}
  \begin{enumerate}[<+->]
    \item User selects resource + start/end time.
    \item System validates inputs (start $<$ end).
    \item System checks overlap for same resource.
    \item System saves booking in DB.
    \item System shows success and updates list view.
  \end{enumerate}
  \vspace{0.2em}
  \textbf{Acceptance criteria (sample):}
  \begin{itemize}
    \item Overlapping booking is rejected with a clear message.
    \item Boundary case: end time equal to another booking's start time is allowed.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Acceptance criteria examples}
  \begin{itemize}[<+->]
    \item ``Booking cannot overlap an existing booking for same resource.''
    \item ``Capacity must be positive integer; else show error.''
    \item ``On success, record is saved in DB and appears in list view.''
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Milestones from use cases}
  \begin{itemize}[<+->]
    \item Group use cases into deliverables by week/session.
    \item Convert each use case into GitHub issues (one issue per feature).
    \item Define ``done'' using acceptance criteria.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}

\begin{frame}{Deliverable-1 checklist}
  \begin{itemize}[<+->]
    \item Use case diagram: 2--3 actors, 6--10 use cases.
    \item Written use cases: 3 core use cases with success + failure flows.
    \item Updated backlog: issues created/updated to match use cases.
  \end{itemize}
  \slidesources{\OOPSUnitVISources}
\end{frame}
